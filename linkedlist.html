add() 
// Adds a node containing 'obj' to the end of the list
add(obj) { 
    const d = this._dummy;
    const n = new Node(obj, d, d.prev);

    d.prev.next = n; 
    d.prev = n; 
    ++this._size; 

    return true;
}
remove() 
// Removes the node containing 'obj';
// returns true if successful
remove(obj) {
    let n = this._dummy.next;
    
    while (n !== this._dummy) {
        if (n.data === object) {
            n.next.prev = n.prev;
            n.prev.next = n.next;
            --this._size;
            return true; 
        }
        
        n = n.next;
    }
    
    return false; 
}
get() 
// Returns element at 'index' without removing
get(index) {
    if (index > this._size - 1)
        throw new Error('Invalid index');

    let n = this._dummy.next;
    for (let i = 0; i < index; ++i) 
        n=n.next;
    
    return n.data;
}
